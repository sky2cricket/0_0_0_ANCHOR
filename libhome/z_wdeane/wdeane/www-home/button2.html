// button2.java
//
// uses buttons to select background color

import java.applet.*;
import java.awt.*;

public class button2 extends Applet
{
 public void init()
 {
  add(new Button("Red"));
  add(new Button("Orange"));
  add(new Button("Yellow"));
  add(new Button("Green"));
  add(new Button("Blue"));
 }

 public boolean action(Event evt, Object whatAction)
 {
  if(!(evt.target instanceof Button))
    return false;

  String buttonLabel = (String) whatAction;

  if(buttonLabel == "Red") setBackground(Color.red);
  if(buttonLabel == "Orange") setBackground(Color.orange);
  if(buttonLabel == "Yellow") setBackground(Color.yellow);
  if(buttonLabel == "Green") setBackground(Color.green);
  if(buttonLabel == "Blue") setBackground(Color.blue);

  repaint();
  return true;
 }
} // end
