Click inside white area. Type any alpha/numeric/punctuation character on keyboard and it's corresponding ASCII value will be displayed.




Source Code 

// keyascii.java
//
// tells you what letters/numbers/symbols you are typing
// and what the cooresponding ASCII value is

import java.awt.*;
import java.applet.*;

public class keyascii extends Applet
{
 String KeyDownEvent = null;

 public void init()
 {
  setBackground(Color.white);
 }

 public boolean keyDown(Event event, int letter)
 {
  KeyDownEvent = "Keyboard = " + (char)letter + "     " + "ASCII value = " + letter;

  repaint();
  return(true);
 }

 public void paint(Graphics g)
 {
  Font a = new Font("TimesRoman", Font.PLAIN, 36);
  g.setFont(a);
  g.setColor(Color.blue);

  if(KeyDownEvent != null)
    g.drawString(KeyDownEvent, 5, 75);
 }
} // end
